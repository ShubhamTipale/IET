 begin
	declare _id int;
	select max(id)+1 into _id from deptLOG;

drop function if exists sumSalary;
delimiter $
create function sumSalary(_deptno int) returns FLOAT
deterministic
BEGIN
	declare sum int default 0;
	select sum(sal) into sum from emp where deptno=_deptno;
	
	IF sum IS NULL THEN
	select "Department not found" AS message;
	END IF;
	
	return sum;
end $
delimiter ;
